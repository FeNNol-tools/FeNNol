 28    The underlying model is a flax.nn.Sequential built from the `modules` dictionary
 29    which references registered modules in `modules.MODULES` and provides the parameters for initialization.
 30
 31    Since the model is static and contains variables, it must be initialized right away with either
 32    `example_data` or `variables`. If `variables` is provided, it is used directly. If `example_data`
 33    is provided, the model is initialized with `example_data` and the resulting variables are stored
 34    in the wrapper.

 37        * __call__  : calls the jitted apply method of the underlying model and sums the energy terms
 38                        in `energy_terms` provided in the constructor.

 40

153    def set_energy_terms(self, energy_terms: Sequence[str] | None, jit=True) -> None:
154        object.__setattr__(self, "energy_terms", energy_terms)
155        if energy_terms is None:
156            def total_energy(variables, data):
157                out = self.__apply(variables, data)
158                coords = out["coordinates"]
159                nsys = out["natoms"].shape[0]
160                nat = coords.shape[0]
161                dtype = coords.dtype
162                e=jnp.zeros(nsys,dtype=dtype)
163                eat = jnp.zeros(nat,dtype=dtype)
164                out["total_energy"] = e
165                out["atomic_energies"] = eat
166                return e, out
167            
168            def energy_and_forces(variables, data):
169                e,out = total_energy(variables, data)
170                f=jnp.zeros_like(out["coordinates"])
171                out["forces"] = f
172                return e,f, out
173            
174            def energy_and_forces_and_virial(variables, data):
175                e,f,out = energy_and_forces(variables, data)
176                v=jnp.zeros((out["natoms"].shape[0],3,3),dtype=out["coordinates"].dtype)
177                out["virial_tensor"] = v
178                return e,f,v, out
179        else:
180            # build the energy and force functions
181            def total_energy(variables, data):
182                out = self.__apply(variables, data)
183                atomic_energies = 0.0
184                system_energies = 0.0
185                species = out["species"]
186                nsys = out["natoms"].shape[0]
187                for term in self.energy_terms:
188                    e = out[term]
189                    if e.ndim > 1 and e.shape[-1] == 1:
190                        e = jnp.squeeze(e, axis=-1)
191                    if e.shape[0] == nsys and nsys != species.shape[0]:
192                        system_energies += e
193                        continue
194                    assert e.shape == species.shape
195                    atomic_energies += e
196                # atomic_energies = jnp.squeeze(atomic_energies, axis=-1)
197                if isinstance(atomic_energies, jnp.ndarray):
198                    if "true_atoms" in out:
199                        atomic_energies = jnp.where(out["true_atoms"], atomic_energies, 0.0)
200                    out["atomic_energies"] = atomic_energies
201                    energies = jax.ops.segment_sum(
202                        atomic_energies,
203                        data["batch_index"],
204                        num_segments=len(data["natoms"]),
205                    )
206                else:
207                    energies = 0.0
208
209                if isinstance(system_energies, jnp.ndarray):
210                    if "true_sys" in out:
211                        system_energies = jnp.where(out["true_sys"], system_energies, 0.0)
212                    out["system_energies"] = system_energies
213
214                out["total_energy"] = energies + system_energies
215                return out["total_energy"], out
216
217            def energy_and_forces(variables, data):
218                def _etot(variables, coordinates):
219                    energy, out = total_energy(
220                        variables, {**data, "coordinates": coordinates}
221                    )
222                    return energy.sum(), out
223
224                de, out = jax.grad(_etot, argnums=1, has_aux=True)(
225                    variables, data["coordinates"]
226                )
227                out["forces"] = -de
228
229                return out["total_energy"], out["forces"], out
230
231            def energy_and_forces_and_virial(variables, data):
232                assert "cells" in data
233                cells = data["cells"]
234                batch_index = data["batch_index"]
235                x = data["coordinates"]
236
237                def _etot(variables, coordinates, cells):
238                    reciprocal_cells = jnp.linalg.inv(cells)
239                    energy, out = total_energy(
240                        variables,
241                        {
242                            **data,
243                            "coordinates": coordinates,
244                            "cells": cells,
245                            "reciprocal_cells": reciprocal_cells,
246                        },
247                    )
248                    return energy.sum(), out
249
250                (dedx, dedcells), out = jax.grad(_etot, argnums=(1, 2), has_aux=True)(
251                    variables, x, cells
252                )
253                # dedx = jnp.einsum("sij,si->sj", reciprocal_cells[batch_index], deds)
254                out["forces"] = -dedx
255                fx = jax.ops.segment_sum(
256                    dedx[:, :, None] * x[:, None, :],
257                    batch_index,
258                    num_segments=len(data["natoms"]),
259                )
260
261                # out["virial_tensor"] = (
262                #     jnp.einsum("sik,skj->sij", dedcells, cells) + fx
263                # )
264                out["virial_tensor"] = jax.vmap(jnp.matmul)(dedcells, cells) + fx
265
266                return out["total_energy"], out["forces"], out["virial_tensor"], out
267
268        if jit:

271            object.__setattr__(

308    def __call__(self, variables: Optional[dict] = None, **inputs) -> Dict[str, Any]:
309        """Apply the FENNIX model (preprocess + modules)
310
311        !!! This is not a pure function => do not apply jax transforms !!!
312        if you want to apply jax transforms, use  self._apply(variables, inputs) which is pure and preprocess the input using self.preprocess
313        """
314        if variables is None:
315            variables = self.variables
316        inputs = self.preprocess(**inputs)
317        output = self._apply(variables, inputs)
318        if self.use_atom_padding:
319            output = atom_unpadding(output)
320        return output

379    def __setattr__(self, __name: str, __value: Any) -> None:
380        if __name == "variables":
381            if __value is not None:
382                if not (
383                    isinstance(__value, dict)
384                    or isinstance(__value, OrderedDict)
385                    or isinstance(__value, FrozenDict)

387                    raise ValueError(f"{__name} must be a dict")

402
403        elif self._initializing:
404            object.__setattr__(self, __name, __value)
405        else:
406            raise ValueError(f"{__name} attribute of FENNIX model is immutable.")

408    def generate_dummy_system(
409        self, rng_key: jax.random.PRNGKey, box_size=None, n_atoms: int = 10
410    ) -> Dict[str, Any]:
411        """
412        Generate dummy system for initialization
413        """
414        if box_size is None:
415            box_size = 2 * self.cutoff
416            for g in self._graphs_properties.values():
417                box_size = min(box_size, 2 * g["cutoff"])
418        coordinates = np.array(
419            jax.random.uniform(rng_key, (n_atoms, 3), maxval=box_size)
420        )
421        species = np.ones((n_atoms,), dtype=np.int32)
422        batch_index = np.zeros((n_atoms,), dtype=np.int32)
423        natoms = np.array([n_atoms], dtype=np.int32)
424        return {
425            "species": species,
426            "coordinates": coordinates,
427            # "graph": graph,
428            "batch_index": batch_index,
429            "natoms": natoms,
430        }

435        if rng_key is None:

Static wrapper for FENNIX models

The underlying model is a flax.nn.Sequential built from the modules dictionary which references registered modules in modules.MODULES and provides the parameters for initialization.

Since the model is static and contains variables, it must be initialized right away with either example_data or variables. If variables is provided, it is used directly. If example_data is provided, the model is initialized with example_data and the resulting variables are stored in the wrapper.

methods: * __call__ : calls the jitted apply method of the underlying model and sums the energy terms in energy_terms provided in the constructor. It returns the total energy (of each subsystem in the batch) and the output dict.

* get_model : returns the underlying model and variables (useful for training)
