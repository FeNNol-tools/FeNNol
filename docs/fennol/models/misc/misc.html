186class Split(nn.Module):
187    key: str
188    output_keys: Sequence[str]
189    axis: int = -1
190    sizes: Union[int, Sequence[int]] = 1
191    squeeze: bool = True
192
193    FID: str = "SPLIT"
194
195    @nn.compact
196    def __call__(self, inputs) -> Any:
197        x = inputs[self.key]
198
199        if isinstance(self.sizes, int):
200            split_size = [self.sizes] * len(self.output_keys)
201        else:
202            split_size = self.sizes
203        if len(split_size) == len(self.output_keys):
204            assert (
205                sum(split_size) == x.shape[self.axis]
206            ), f"Split sizes {split_size} do not match input shape"
207            split_size = split_size[:-1]
208        assert (
209            len(split_size) == len(self.output_keys) - 1
210        ), f"Wrong number of split sizes {split_size} for {len(self.output_keys)} outputs"
211        split_indices = np.cumsum(split_size)
212        outs = {}
213
214        for k, v in zip(self.output_keys, jnp.split(x, split_indices, axis=self.axis)):
215            outs[k] = (
216                jnp.squeeze(v, axis=self.axis)
217                if self.squeeze and v.shape[self.axis] == 1
218                else v
219            )
220
221        return {**inputs, **outs}

372class SwitchFunction(nn.Module):
373    cutoff: Optional[float] = None
374    switch_start: float = 0.0
375    graph_key: Optional[str] = "graph"
376    output_key: Optional[str] = None
377    switch_type: str = "cosine"
378    p: Optional[float] = None
379    trainable: bool = False
380
381    FID: str = "SWITCH_FUNCTION"
382
383
384    @nn.compact
385    def __call__(self, inputs) -> Any:
386        if self.graph_key is not None:
387            graph = inputs[self.graph_key]
388            distances, edge_mask = graph["distances"], graph["edge_mask"]
389            if self.cutoff is not None:
390                edge_mask = jnp.logical_and(edge_mask, (distances < self.cutoff))
391                cutoff = self.cutoff
392            else:
393                cutoff = graph["cutoff"]
394        else:
395            # distances = inputs
396            distances, edge_mask = inputs
397            assert (
398                self.cutoff is not None
399            ), "cutoff must be specified if no graph is given"
400            # edge_mask = distances < self.cutoff
401            cutoff = self.cutoff
402
403        if self.switch_start > 1.0e-5:
404            assert (
405                self.switch_start < 1.0
406            ), "switch_start is a proportion of cutoff and must be smaller than 1."
407            cutoff_in = self.switch_start * cutoff
408            x = distances - cutoff_in
409            end = cutoff - cutoff_in
410        else:
411            x = distances
412            end = cutoff
413
414        switch_type = self.switch_type.lower()
415        if switch_type == "cosine":
416            p = self.p if self.p is not None else 1.0
417            if self.trainable:
418                p = self.param("p", lambda rng: jnp.asarray(p, dtype=jnp.float32))
419            switch = (0.5 * jnp.cos(x * (jnp.pi / end)) + 0.5) ** p
420
421        elif switch_type == "polynomial":
422            p = self.p if self.p is not None else 3.0
423            if self.trainable:
424                p = self.param("p", lambda rng: jnp.asarray(p, dtype=jnp.float32))
425            d = x / end
426            switch = (
427                1.0
428                - 0.5 * (p + 1) * (p + 2) * d**p
429                + p * (p + 2) * d ** (p + 1)
430                - 0.5 * p * (p + 1) * d ** (p + 2)
431            )
432
433        elif switch_type == "exponential":
434            p = self.p if self.p is not None else 1.0
435            if self.trainable:
436                p = self.param("p", lambda rng: jnp.asarray(p, dtype=jnp.float32))
437            r2 = x**2
438            c2 = end**2
439            switch = jnp.exp(-p * r2 / (c2 - r2))
440
441        else:
442            raise ValueError(f"Unknown switch function {switch_type}")
443
444        if self.switch_start > 1.0e-5:
445            switch = jnp.where(distances < cutoff_in, 1.0, switch)
446
447        switch = jnp.where(edge_mask, switch, 0.0)
448
449        if self.graph_key is not None:
450            if self.output_key is not None:
451                return {**inputs, self.output_key: switch}
452            else:
453                return {**inputs, self.graph_key: {**graph, "switch": switch}}
454        else:
455            return switch #, edge_mask

Wraps parent module references in weak refs.

This prevents reference cycles from forming via parent links which can lead to accidental OOMs in eager mode due to slow garbage collection as well as spurious tracer leaks during jit compilation.

Note: "descriptors" are the underlying python mechanism for implementing dynamic @property decorators. We need to use a raw descriptor instead of the more common decorator in order to force that the appropriate getter/setter logic applies in subclasses even after various dataclass transforms.

385    @nn.compact
386    def __call__(self, inputs) -> Any:
387        if self.graph_key is not None:
388            graph = inputs[self.graph_key]
389            distances, edge_mask = graph["distances"], graph["edge_mask"]
390            if self.cutoff is not None:
391                edge_mask = jnp.logical_and(edge_mask, (distances < self.cutoff))
392                cutoff = self.cutoff
393            else:
394                cutoff = graph["cutoff"]
395        else:
396            # distances = inputs
397            distances, edge_mask = inputs
398            assert (
399                self.cutoff is not None
400            ), "cutoff must be specified if no graph is given"
401            # edge_mask = distances < self.cutoff
402            cutoff = self.cutoff
403
404        if self.switch_start > 1.0e-5:
405            assert (
406                self.switch_start < 1.0
407            ), "switch_start is a proportion of cutoff and must be smaller than 1."
408            cutoff_in = self.switch_start * cutoff
409            x = distances - cutoff_in
410            end = cutoff - cutoff_in
411        else:
412            x = distances
413            end = cutoff
414
415        switch_type = self.switch_type.lower()
416        if switch_type == "cosine":
417            p = self.p if self.p is not None else 1.0
418            if self.trainable:
419                p = self.param("p", lambda rng: jnp.asarray(p, dtype=jnp.float32))
420            switch = (0.5 * jnp.cos(x * (jnp.pi / end)) + 0.5) ** p
421
422        elif switch_type == "polynomial":
423            p = self.p if self.p is not None else 3.0
424            if self.trainable:
425                p = self.param("p", lambda rng: jnp.asarray(p, dtype=jnp.float32))
426            d = x / end
427            switch = (
428                1.0
429                - 0.5 * (p + 1) * (p + 2) * d**p
430                + p * (p + 2) * d ** (p + 1)
431                - 0.5 * p * (p + 1) * d ** (p + 2)
432            )
433
434        elif switch_type == "exponential":
435            p = self.p if self.p is not None else 1.0
436            if self.trainable:
437                p = self.param("p", lambda rng: jnp.asarray(p, dtype=jnp.float32))
438            r2 = x**2
439            c2 = end**2
440            switch = jnp.exp(-p * r2 / (c2 - r2))
441
442        else:
443            raise ValueError(f"Unknown switch function {switch_type}")
444
445        if self.switch_start > 1.0e-5:
446            switch = jnp.where(distances < cutoff_in, 1.0, switch)
447
448        switch = jnp.where(edge_mask, switch, 0.0)
449
450        if self.graph_key is not None:
451            if self.output_key is not None:
452                return {**inputs, self.output_key: switch}
453            else:
454                return {**inputs, self.graph_key: {**graph, "switch": switch}}
455        else:
456            return switch #, edge_mask
